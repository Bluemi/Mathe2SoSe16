		\frac{\frac{1}{g(x)} - \frac{1}{f(x)}}{\frac{1}{f(x) \ g(x)}} && 	
		\text{0/0}
	\\	
	\beta)& \ falls \lim_{x \to x_0} \frac{g(x)}{f(x)} = 1, \lim_{x \to x_0}
		\big(f(x)-g(x)\big)=\lim_{x \to x_0}\bigg[f(x)(1-\frac{g(x)}{f(x)}) \bigg]
	\\
	\delta)& \ f(x) - g(x) = \ln e^{f(x)-g(x)} = ln\frac{e^{f(x)}}{e^{g(x)}} &&
