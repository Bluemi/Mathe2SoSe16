\begin{align*}
	f(x)=&-f(-x)\\
	f'(x)=&(x^{-1})'=-x^{-2}=-\frac{1}{x^2}
\end{align*}

\paragraph{Fallunterscheidung}
\begin{enumerate}
\item $x>0:|x|=x$
	\begin{align*}
		y=&f(x)=\ln|x|=\ln x\\
		x=&f^{-1}(y)=g(y)=e^y\\
		\Rightarrow& g'(y)=e^y\\
		1=&(g \circ f)'(x)=g'(y)\bigg \vert _{y=f(x)} f'(x)\\ 
		\Rightarrow& [\ln x]'=f'(x)\\
		=&\frac{1}{g'(y)\bigg \vert _{y=f(x)}}\\
		=&\frac{1}{e^y\bigg \vert _{y=\ln x}}\\
		=&\frac{1}{x}
	\end{align*}
\item $x<0:|x|=-x$
	\begin{align*}
		y=&f(x)=\ln |x|=\ln(-x)=\ln(h(x))\\
		% ist das wirklich nen ln nach dem z??
		z=&\ln(x)=-x>0, h'(x)=-1\\
		f'(x)=&(\ln \circ h)'(x)=[ln z]\bigg \vert _{z=h(x)} , h'(x)\\ % wirklich ein komma??
		=&\frac{1}{z}\bigg \vert _{z=-x} , (-1) = \frac{-1}{-x}=\frac{1}{x}
	\end{align*}
\end{enumerate}
